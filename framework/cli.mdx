<Steps>
<Step title="Installing Composio">
  <CodeGroup>
  ```bash Installing via pip
  pip install composio
  ```
  </CodeGroup>
  Install Package For Selected Framework
  <AccordionGroup>
  <Accordion title="Langchain">
    <CodeGroup>
      ```bash Installation for Langchain
      pip install composio_langchain
      ```
    </CodeGroup>
  </Accordion>
  <Accordion title="Autogen">
    <CodeGroup>
      ```bash Installation for Autogen
      pip install composio_autogen
      ```
    </CodeGroup>
  </Accordion>
  <Accordion title="CrewAI">
    <CodeGroup>
      ```bash Installation for CrewAI
      pip install composio_crewai
      ```
    </CodeGroup>
  </Accordion>
  </AccordionGroup>
  </Step>


<Step title="Check Supported Apps">
<CodeGroup>
```bash Print Supported Apps
composio-cli show-apps
```
</CodeGroup>
</Step>
<Step title="Connect Selected App">
<CodeGroup>
```bash Connect Github App
composio-cli add github
```
</CodeGroup>
</Step>
<Step title="Update Actions & Integrations">
<CodeGroup>
```bash Refresh the Integrations & Actions Index
composio-cli update
```
</CodeGroup>
</Step>
</Steps>
