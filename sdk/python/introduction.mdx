  <AccordionGroup>
  <Accordion title="Langchain">

    </CodeGroup>
  </Accordion>
  <Accordion title="Autogen">
    <CodeGroup>
      ```bash Installation for Autogen
      pip install composio-autogen
      ```
    </CodeGroup>
  </Accordion>
  <Accordion title="CrewAI">
    <CodeGroup>
      ```bash Installation for CrewAI
      pip install composio-crewai
      ```
    </CodeGroup>
  </Accordion>
  </AccordionGroup>
</Step>
